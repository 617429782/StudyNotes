语法
  background: color image repeat attachment position / size origin clip;
  中间的斜杠不能省略，例: background: url(http://xxx.jpg) no-repeat center / 100%;

取值: 
  color: transparent [ white | #fff| rgb(255,255,255) ]
  image: none []
  repeat: repeat []
  attachment: scroll [ fixed ] 
  position: 0% 0% []
    
  size: auto auto
  origin: padding-box
  clip: border-box
